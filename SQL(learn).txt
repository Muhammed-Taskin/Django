SELECT * FROM tablo_ismi;// Bana tablodaki tüm verileri ver demektir(*)
SELECT * FROM tablo_ismi WHERE id=7 ;// bana tabloda id'si 7 olan veriyi sunuyor.
SELECT * FROM tablo_ismi WHERE name="twitter"; // bana name kısmında twitter yazan veriyi getirdi.

## FILTRELEME ##

SELECT * FROM tablo_ismi WHERE LIKE "%SQL%"; // İçinde sql yazan verileri filtreledik

***!!! VERİ SİLME VE GÜNCELLEME KOMUTLARI !!!***

UPDATE tablo_ismi SET name="MUHAMMMED" WHERE id=7;
// Bana id'si 7 olan veriyi name kısmında muhammed yazdır demektir.

DELETE FROM tablo_ismi WHERE id=7;
// Bana id'si 7 olan veriyi sil demektir.
